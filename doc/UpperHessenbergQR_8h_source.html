67  virtual void compute(const Matrix &mat)
68  {
69  n = mat.n_rows;
70  mat_T.set_size(n, n);
71  rot_cos.set_size(n - 1);
72  rot_sin.set_size(n - 1);
73 
74  mat_T = arma::trimatu(mat);
75  mat_T.diag(-1) = mat.diag(-1);
76 
77  Scalar xi, xj, r, c, s;
78  Matrix Gt(2, 2);
79  for(int i = 0; i < n - 2; i++)
80  {
81  xi = mat_T(i, i);
82  xj = mat_T(i + 1, i);
83  r = std::sqrt(xi * xi + xj * xj);
84  rot_cos[i] = c = xi / r;
85  rot_sin[i] = s = -xj / r;
86  // For a complete QR decomposition,
87  // we first obtain the rotation matrix
88  // G = [ cos sin]
89  // [-sin cos]
90  // and then do T[i:(i + 1), i:(n - 1)] = G' * T[i:(i + 1), i:(n - 1)]
91 
92  Gt << c << -s << arma::endr << s << c << arma::endr;
93  mat_T.submat(i, i, i + 1, n - 1) = Gt * mat_T.submat(i, i, i + 1, n - 1);
94 
95  // If we do not need to calculate the R matrix, then
96  // only the cos and sin sequences are required.
97  // In such case we only update T[i + 1, (i + 1):(n - 1)]
98  // mat_T(i + 1, arma::span(i + 1, n - 1)) *= c;
99  // mat_T(i + 1, arma::span(i + 1, n - 1)) += s * mat_T(i, arma::span(i + 1, n - 1));
100  }
101  // For i = n - 2
102  xi = mat_T(n - 2, n - 2);
103  xj = mat_T(n - 1, n - 2);
104  r = std::sqrt(xi * xi + xj * xj);
105  rot_cos[n - 2] = c = xi / r;
106  rot_sin[n - 2] = s = -xj / r;
107  Gt << c << -s << arma::endr << s << c << arma::endr;
108  mat_T.submat(n - 2, n - 2, n - 1, n - 1) = Gt * mat_T.submat(n - 2, n - 2, n - 1, n - 1);
109 
110  computed = true;
111  }

385  void compute(const Matrix &mat)
386  {
387  this->n = mat.n_rows;
388  this->mat_T.set_size(this->n, this->n);
389  this->rot_cos.set_size(this->n - 1);
390  this->rot_sin.set_size(this->n - 1);
391 
392  this->mat_T.zeros();
393  this->mat_T.diag() = mat.diag();
394  this->mat_T.diag(1) = mat.diag(-1);
395  this->mat_T.diag(-1) = mat.diag(-1);
396 
397  // A number of pointers to avoid repeated address calculation
398  Scalar *Tii = this->mat_T.memptr(), // pointer to T[i, i]
399  *ptr, // some location relative to Tii
400  *c = this->rot_cos.memptr(), // pointer to the cosine vector
401  *s = this->rot_sin.memptr(), // pointer to the sine vector
402  r, tmp;
403  for(int i = 0; i < this->n - 2; i++)
404  {
405  // Tii[0] == T[i, i]
406  // Tii[1] == T[i + 1, i]
407  r = std::sqrt(Tii[0] * Tii[0] + Tii[1] * Tii[1]);
408  *c = Tii[0] / r;
409  *s = -Tii[1] / r;
410 
411  // For a complete QR decomposition,
412  // we first obtain the rotation matrix
413  // G = [ cos sin]
414  // [-sin cos]
415  // and then do T[i:(i + 1), i:(i + 2)] = G' * T[i:(i + 1), i:(i + 2)]
416 
417  // Update T[i, i] and T[i + 1, i]
418  // The updated value of T[i, i] is known to be r
419  // The updated value of T[i + 1, i] is known to be 0
420  Tii[0] = r;
421  Tii[1] = 0;
422  // Update T[i, i + 1] and T[i + 1, i + 1]
423  // ptr[0] == T[i, i + 1]
424  // ptr[1] == T[i + 1, i + 1]
425  ptr = Tii + this->n;
426  tmp = *ptr;
427  ptr[0] = (*c) * tmp - (*s) * ptr[1];
428  ptr[1] = (*s) * tmp + (*c) * ptr[1];
429  // Update T[i, i + 2] and T[i + 1, i + 2]
430  // ptr[0] == T[i, i + 2] == 0
431  // ptr[1] == T[i + 1, i + 2]
432  ptr += this->n;
433  ptr[0] = -(*s) * ptr[1];
434  ptr[1] *= (*c);
435 
436  // Move from T[i, i] to T[i + 1, i + 1]
437  Tii += this->n + 1;
438  // Increase c and s by 1
439  c++;
440  s++;
441 
442 
443  // If we do not need to calculate the R matrix, then
444  // only the cos and sin sequences are required.
445  // In such case we only update T[i + 1, (i + 1):(i + 2)]
446  // this->mat_T(i + 1, i + 1) = (*c) * this->mat_T(i + 1, i + 1) + (*s) * this->mat_T(i, i + 1);
447  // this->mat_T(i + 1, i + 2) *= (*c);
448  }
449  // For i = n - 2
450  r = std::sqrt(Tii[0] * Tii[0] + Tii[1] * Tii[1]);
451  *c = Tii[0] / r;
452  *s = -Tii[1] / r;
453  Tii[0] = r;
454  Tii[1] = 0;
455  ptr = Tii + this->n; // points to T[i - 2, i - 1]
456  tmp = *ptr;
457  ptr[0] = (*c) * tmp - (*s) * ptr[1];
458  ptr[1] = (*s) * tmp + (*c) * ptr[1];
459 
460  this->computed = true;
